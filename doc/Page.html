def self.find_or_create(url)
  return nil unless PAGE_LIST.include?(url)
  return Page.find_by_url(url) ||
         Page.create(url: url,
              name: PAGE_LIST[url],
              message: "This page is blank.
              \r\n\r\n>_You can **edit** this page to add new content._")
end
